1. Token 与分词原理

LLM 看到的不是"字"，而是 token。英文 1 token ≈ 0.75 词，中文 1 字 ≈ 1.5-2 tokens。这就是为什么中文 API 调用更贵。

# Demo: Token 可视化工具
import tiktoken

def token_demo():
    enc = tiktoken.encoding_for_model("gpt-4o")

    texts = {
        "英文": "The quick brown fox jumps over the lazy dog.",
        "中文": "敏捷的棕色狐狸跳过了那只懒狗。",
        "代码": "def fibonacci(n):\n    if n <= 1: return n\n    return fibonacci(n-1) + fibonacci(n-2)",
        "混合": "请帮我写一个 Python 函数来计算 Fibonacci 数列。"
    }

    for label, text in texts.items():
        tokens = enc.encode(text)
        decoded = [enc.decode([t]) for t in tokens]
        print(f"\n{'='*60}")
        print(f"【{label}】字符数: {len(text)} | Token 数: {len(tokens)} | 比例: {len(tokens)/len(text):.2f}")
        print(f"Token 列表: {tokens[:15]}{'...' if len(tokens)>15 else ''}")
        print(f"解码预览: {''.join(decoded[:10])}...")

    # 成本计算
    prices = {"gpt-4o": (2.50, 10.00), "gpt-4o-mini": (0.15, 0.60),
              "claude-opus-4": (15.00, 75.00), "deepseek-v3": (0.27, 1.10)}
    input_tokens, output_tokens = 1000, 500
    for model, (in_price, out_price) in prices.items():
        cost = (input_tokens/1_000_000)*in_price + (output_tokens/1_000_000)*out_price
        print(f"  {model:20s}: ${cost:.4f} / 1K in + 500 out")

2. Embedding 语义向量

Embedding 将文本映射到高维空间中的点，语义相似的文本在空间中距离近。

# Demo: 迷你语义搜索引擎
import numpy as np
from openai import OpenAI
client = OpenAI()

def get_embedding(text: str, model="text-embedding-3-small") -> list[float]:
    return client.embeddings.create(input=[text], model=model).data[0].embedding

def cosine_similarity(a: list[float], b: list[float]) -> float:
    a_np, b_np = np.array(a), np.array(b)
    return float(np.dot(a_np, b_np) / (np.linalg.norm(a_np) * np.linalg.norm(b_np)))

# Demo: 统一 LLM 客户端 — 一行代码切换模型
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import AsyncIterator
from openai import AsyncOpenAI
from anthropic import AsyncAnthropic
import os

@dataclass
class LLMResponse:
    content: str
    model: str
    input_tokens: int
    output_tokens: int
    duration_ms: float

class BaseLLMClient(ABC):
    @abstractmethod
    async def chat(self, messages: list[dict], **kwargs) -> LLMResponse: ...
    @abstractmethod
    async def chat_stream(self, messages: list[dict], **kwargs) -> AsyncIterator[str]: ...

class OpenAIClient(BaseLLMClient):
    def __init__(self, model="gpt-4o-mini"):
        self.model = model
        self.client = AsyncOpenAI(api_key=os.getenv("OPENAI_API_KEY"))

    async def chat(self, messages, **kwargs):
        import time; start = time.perf_counter()
        resp = await self.client.chat.completions.create(model=self.model, messages=messages, **kwargs)
        elapsed = (time.perf_counter() - start) * 1000
        return LLMResponse(content=resp.choices[0].message.content, model=self.model,
                          input_tokens=resp.usage.prompt_tokens, output_tokens=resp.usage.completion_tokens,
                          duration_ms=elapsed)

    async def chat_stream(self, messages, **kwargs):
        stream = await self.client.chat.completions.create(model=self.model, messages=messages, stream=True, **kwargs)
        async for chunk in stream:
            if chunk.choices[0].delta.content:
                yield chunk.choices[0].delta.content

class AnthropicClient(BaseLLMClient):
    def __init__(self, model="claude-sonnet-4-6"):
        self.model = model
        self.client = AsyncAnthropic(api_key=os.getenv("ANTHROPIC_API_KEY"))

    async def chat(self, messages, **kwargs):
        import time; start = time.perf_counter()
        system = next((m["content"] for m in messages if m["role"]=="system"), "You are helpful.")
        user_msgs = [m for m in messages if m["role"]!="system"]
        resp = await self.client.messages.create(model=self.model, system=system, messages=user_msgs,
                                                  max_tokens=kwargs.get("max_tokens", 4096))
        elapsed = (time.perf_counter() - start) * 1000
        return LLMResponse(content=resp.content[0].text, model=self.model,
                          input_tokens=resp.usage.input_tokens, output_tokens=resp.usage.output_tokens,
                          duration_ms=elapsed)

    async def chat_stream(self, messages, **kwargs):
        system = next((m["content"] for m in messages if m["role"]=="system"), "You are helpful.")
        user_msgs = [m for m in messages if m["role"]!="system"]
        async with self.client.messages.stream(model=self.model, system=system, messages=user_msgs,
                                                max_tokens=kwargs.get("max_tokens", 4096)) as stream:
            async for text in stream.text_stream:
                yield text

# Demo: 150 行实现完整 RAG 系统
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_community.document_loaders import PyPDFLoader, TextLoader, UnstructuredMarkdownLoader
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.runnables import RunnablePassthrough
from langchain_core.output_parsers import StrOutputParser
import os

class SimpleRAG:
    """最小化 RAG 系统 — 清晰展示每个环节"""

    def __init__(self, persist_dir="./chroma_db"):
        # 1. Embedding 模型: 文本 → 向量
        self.embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
        # 2. LLM: 生成最终答案
        self.llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
        # 3. 向量数据库: 存储和检索
        self.vectorstore = None
        self.persist_dir = persist_dir

    def ingest_documents(self, file_paths: list[str]) -> int:
        """离线阶段: 加载文档 → 分块 → 向量化 → 存储"""
        all_docs = []
        for fp in file_paths:
            if fp.endswith('.pdf'):
                loader = PyPDFLoader(fp)
            elif fp.endswith('.md'):
                loader = UnstructuredMarkdownLoader(fp)
            else:
                loader = TextLoader(fp)
            all_docs.extend(loader.load())
            print(f"  加载: {fp} ({len(all_docs)} 页/段)")

        # 文本分割: 块大小 512 tokens, 重叠 50 tokens
        splitter = RecursiveCharacterTextSplitter(
            chunk_size=512, chunk_overlap=50,
            separators=["\n\n", "\n", "。", ".", " ", ""]
        )
        chunks = splitter.split_documents(all_docs)
        print(f"  分块: {len(chunks)} 个文本块")

        # 向量化 + 存入 ChromaDB
        self.vectorstore = Chroma.from_documents(
            documents=chunks, embedding=self.embeddings,
            persist_directory=self.persist_dir
        )
        return len(chunks)

    def query(self, question: str, top_k: int = 5, use_rerank: bool = False) -> dict:
        """在线阶段: 检索 → 重排序 → 生成"""
        if self.vectorstore is None:
            self.vectorstore = Chroma(persist_directory=self.persist_dir,
                                       embedding_function=self.embeddings)

        # Step 1: 基础检索 (向量相似度)
        retriever = self.vectorstore.as_retriever(
            search_type="mmr",  # MMR: 最大边际相关 — 兼顾相关性和多样性
            search_kwargs={"k": top_k, "fetch_k": top_k * 4}
        )
        docs = retriever.invoke(question)

        # Step 2 (可选): Cohere Rerank (需要 COHERE_API_KEY)
        if use_rerank and os.getenv("COHERE_API_KEY"):
            from langchain_cohere import CohereRerank
            reranker = CohereRerank(top_n=3)
            docs = reranker.compress_documents(docs, question)

        # Step 3: 拼接上下文 + Prompt → LLM 生成
        context = "\n\n---\n\n".join(
            f"[来源 {i+1}] {d.page_content}" for i, d in enumerate(docs)
        )

        prompt = ChatPromptTemplate.from_messages([
            ("system", """你是企业知识库助手。严格基于提供的文档内容回答问题。
如果文档中没有相关信息，请明确说"文档中未找到相关信息"。
在回答末尾列出引用的来源编号。"""),
            ("user", "参考资料:\n{context}\n\n问题: {question}")
        ])

        chain = prompt | self.llm | StrOutputParser()
        answer = chain.invoke({"context": context, "question": question})

        return {
            "question": question,
            "answer": answer,
            "sources": [
                {"content": d.page_content[:200] + "...",
                 "metadata": d.metadata}
                for d in docs
            ]
        }

# Demo: 用 RAGAS 评估 RAG 质量
from datasets import Dataset
from ragas import evaluate
from ragas.metrics import (
    faithfulness,        # 答案是否忠实于检索到的文档 (0-1)
    answer_relevancy,    # 答案是否与问题相关 (0-1)
    context_recall,      # 检索到的文档覆盖了多少必要信息
    context_precision,   # 检索到的文档中有多少是真正相关的
)

def evaluate_rag(questions: list[str], answers: list[str],
                 contexts: list[list[str]], ground_truths: list[str] = None):
    """评估 RAG 系统的检索和生成质量"""
    data = {
        "question": questions,
        "answer": answers,
        "contexts": contexts,
    }
    if ground_truths:
        data["ground_truth"] = ground_truths

    dataset = Dataset.from_dict(data)
    metrics = [faithfulness, answer_relevancy, context_recall, context_precision]
    result = evaluate(dataset, metrics=metrics)

    print("RAG 质量评估报告")
    print(f"  Faithfulness (忠实度):      {result['faithfulness']:.3f}")
    print(f"  Answer Relevancy (答案相关): {result['answer_relevancy']:.3f}")
    print(f"  Context Recall (召回率):     {result['context_recall']:.3f}")
    print(f"  Context Precision (精确率):  {result['context_precision']:.3f}")
    return result

from typing import Annotated, Literal
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, END
from langgraph.graph.message import add_messages
from langgraph.checkpoint.memory import MemorySaver
from langgraph.checkpoint.sqlite import SqliteSaver

from langchain_core.messages import HumanMessage, AIMessage, ToolMessage
import json, sqlite3

# ===== 1. 定义状态 =====
class AgentState(TypedDict):
    messages: Annotated[list, add_messages]
    user_id: str
    turn_count: int

# ===== 2. 准备 LLM + Tools =====
llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
llm_with_tools = llm.bind_tools(ALL_TOOLS)

# ===== 3. 定义节点 =====
def agent_node(state: AgentState) -> dict:
    """Agent节点: LLM决策下一步做什么"""
    response = llm_with_tools.invoke(state["messages"])
    return {"messages": [response],
            "turn_count": state.get("turn_count", 0) + 1}

def tool_node(state: AgentState) -> dict:
    """Tool节点: 执行工具调用并返回结果"""
    last_message = state["messages"][-1]
    tool_messages = []
    for tc in last_message.tool_calls:
        tool_name = tc["name"]
        tool_args = tc["args"]
        tool = next(t for t in ALL_TOOLS if t.name == tool_name)
        result = tool.invoke(tool_args)
        tool_messages.append(ToolMessage(
            content=str(result),
            tool_call_id=tc["id"],
            name=tool_name
        ))
    return {"messages": tool_messages}

# ===== 4. 路由函数 =====
def should_continue(state: AgentState) -> Literal["tools", "__end__"]:
    last_message = state["messages"][-1]
    if (hasattr(last_message, "tool_calls")
            and last_message.tool_calls):
        # 敏感操作可在这里中断
        for tc in last_message.tool_calls:
            if tc["name"] == "send_email":
                print(f"需要审批: 发送邮件给 "
                      f"{tc['args'].get('to')}")
        return "tools"
    return "__end__"

# ===== 5. 构建图 =====
def build_agent(checkpointer=None):
    workflow = StateGraph(AgentState)
    workflow.add_node("agent", agent_node)
    workflow.add_node("tools", tool_node)
    workflow.set_entry_point("agent")
    workflow.add_conditional_edges(
        "agent", should_continue,
        {"tools": "tools", "__end__": END}
    )
    workflow.add_edge("tools", "agent")
    return workflow.compile(checkpointer=checkpointer)

                    output_tokens: int) -> dict:

        cost = ((input_tokens / 1_000_000) * in_p +
                (output_tokens / 1_000_000) * out_p)

        self.daily_usage[today] += cost

        if self.daily_usage[today] > self.alert_threshold:
            print(f"ALERT: 今日费用 ${self.daily_usage[today]:.2f} "
                  f"已超过阈值 ${self.alert_threshold}!")

        return {
            "cost": round(cost, 6),
            "daily_total": round(self.daily_usage[today], 2)
        }

    def report(self):
        print("\n=== TOKEN 费用报告 ===")
        for day, cost in sorted(self.daily_usage.items()):
            bar = "#" * int(cost / 2)
            print(f"  {day}: ${cost:.2f} {bar}")

# Demo: 企业级 AI Gateway — 统一入口、认证、限流、成本追踪
from fastapi import FastAPI, HTTPException, Depends, Header
from pydantic import BaseModel
import litellm, time
from collections import defaultdict

app = FastAPI(title="Enterprise AI Gateway", version="1.0.0")

# ==== 数据模型 ====
class ChatRequest(BaseModel):
    model: str = "gpt-4o-mini"
    messages: list[dict]
    temperature: float = 0.7
    max_tokens: int = 4096

class ChatResponse(BaseModel):
    id: str; model: str; content: str
    usage: dict; cost: float; latency_ms: float

# ==== API Key 认证 ====
VALID_KEYS = {"sk-enterprise-001": "tenant-a", "sk-enterprise-002": "tenant-b"}

def verify_api_key(authorization: str = Header(...)):
    api_key = authorization.replace("Bearer ", "")
    if api_key not in VALID_KEYS:
        raise HTTPException(status_code=401, detail="Invalid API Key")
    return VALID_KEYS[api_key]

# ==== Token Bucket 限流 ====
class TokenBucket:
    def __init__(self, rate=60, capacity=120):
        self.rate = rate; self.capacity = capacity
        self.tokens = capacity; self.last_refill = time.time()

    def consume(self, tokens=1) -> bool:
        now = time.time()
        elapsed = now - self.last_refill
        self.tokens = min(self.capacity, self.tokens + elapsed * self.rate)
        self.last_refill = now
        if self.tokens >= tokens:
            self.tokens -= tokens; return True
        return False

rate_limiters: dict[str, TokenBucket] = defaultdict(
    lambda: TokenBucket(rate=60)
)

# ==== 定价表 ====
PRICING = {
    "gpt-4o": (2.50, 10.00), "gpt-4o-mini": (0.15, 0.60),
    "claude-sonnet-4-6": (3.00, 15.00), "claude-opus-4-7": (15.00, 75.00),
    "deepseek-v3": (0.27, 1.10), "gemini-2.5-flash": (0.15, 0.60),
}

def calc_cost(model: str, inp: int, out: int) -> float:
    in_p, out_p = PRICING.get(model, (1.0, 5.0))
    return (inp/1_000_000)*in_p + (out/1_000_000)*out_p

# ==== 核心 API ====
@app.post("/v1/chat/completions", response_model=ChatResponse)
async def chat(req: ChatRequest, tenant=Depends(verify_api_key)):
    # 1. Rate Limit
    if not rate_limiters[tenant].consume(5):
        raise HTTPException(429, detail="Rate limit exceeded")

    # 2. 调用 LLM (LiteLLM 统一接口 + 自动Fallback)
    start = time.perf_counter()
    try:
        response = await litellm.acompletion(
            model=req.model, messages=req.messages,
            temperature=req.temperature, max_tokens=req.max_tokens,
            fallbacks=[{"model": "gpt-4o-mini"}]
        )
    except Exception as e:
        raise HTTPException(500, detail=f"LLM Error: {e}")

    latency = (time.perf_counter() - start) * 1000
    usage = response.usage
    cost = calc_cost(req.model, usage.prompt_tokens,
                     usage.completion_tokens)

    # 3. 审计日志
    print(f"[AUDIT] tenant={tenant} model={req.model} "
          f"in={usage.prompt_tokens} out={usage.completion_tokens} "
          f"cost=${cost:.6f} latency={latency:.0f}ms")

    return ChatResponse(
        id=response.id, model=req.model,
        content=response.choices[0].message.content,
        usage={"input": usage.prompt_tokens,
               "output": usage.completion_tokens},
        cost=cost, latency_ms=latency
    )

  "intent": "faq|order_query|complaint|human",

import os, json
from typing import Optional
from github import Github

class AICodeReviewer:
    """AI 代码审查 Agent: 接收PR -> 分析Diff -> 生成Review"""

    def __init__(self, github_token: str):
        self.gh = Github(github_token)
        self.llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

    def review_pr(self, repo_name: str, pr_number: int) -> list[dict]:
        """审查指定 PR"""
        repo = self.gh.get_repo(repo_name)
        pr = repo.get_pull(pr_number)
        files = pr.get_files()

        reviews = []
        for file in files:
            if file.filename.endswith(('.py', '.js', '.ts', '.java', '.go')):
                result = self._analyze_diff(file.filename, file.patch)
                if result["issues"]:
                    reviews.append(result)

        # 生成审查摘要
        summary = self._generate_summary(reviews)
        return {"summary": summary, "file_reviews": reviews}

    def _analyze_diff(self, filename: str, patch: str) -> dict:
        """分析单个文件的 diff"""
        prompt = f"""你是资深代码审查专家。审查以下代码变更。

文件: {filename}
变更内容:
{patch}

请检查:
1. 安全漏洞 (SQL注入, XSS, 硬编码密钥, 不安全反序列化)
2. 逻辑错误 (空指针, 边界条件, 竞态条件)
3. 性能问题 (N+1查询, 不必要的循环, 内存泄漏)
4. 代码规范 (命名, 注释, SOLID原则)

输出JSON格式:
{{
  "issues": [
    {{"severity": "critical|high|medium|low",
      "line": 行号,
      "category": "类别",
      "description": "问题描述",
      "suggestion": "修改建议"}}
  ],
  "overall_score": 1-10
}}
只输出JSON, 如果没问题返回 {{"issues":[], "overall_score":10}}"""

        resp = self.llm.invoke(prompt)
        try:
            return json.loads(resp.content)
        except json.JSONDecodeError:
            return {"issues": [], "overall_score": 0, "error": "解析失败"}

    def _generate_summary(self, reviews: list[dict]) -> str:
        """生成审查总结"""
        all_issues = []
        for r in reviews:
            all_issues.extend(r.get("issues", []))

        critical = sum(1 for i in all_issues if i["severity"] == "critical")
        high = sum(1 for i in all_issues if i["severity"] == "high")

        return (f"代码审查完成: 发现 {len(all_issues)} 个问题 "
                f"({critical} 严重, {high} 高危)")

# Demo: Text-to-SQL Agent — 自然语言 -> SQL -> 执行 -> 可视化
from langchain_openai import ChatOpenAI
import sqlite3

class TextToSQLAgent:
    """自然语言转 SQL 数据分析 Agent"""

    def __init__(self, db_path: str):
        self.llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
        self.conn = sqlite3.connect(db_path)
        self.schema = self._get_schema()

    def _get_schema(self) -> str:
        """读取数据库所有表结构"""
        cursor = self.conn.cursor()
        tables = cursor.execute(
            "SELECT name FROM sqlite_master WHERE type='table'"
        ).fetchall()

        schema_parts = []
        for (table,) in tables:
            cols = cursor.execute(f"PRAGMA table_info({table})").fetchall()
            cols_str = ", ".join(
                f"{c[1]} {c[2]}" for c in cols
            )
            schema_parts.append(f"  {table}({cols_str})")
        return "\n".join(schema_parts)

    def query(self, question: str) -> dict:
        """自然语言 -> SQL -> 执行 -> 结果"""
        prompt = f"""你是SQL专家。数据库Schema:
{self.schema}

用户问题: {question}

规则:
1. 只生成SELECT语句
2. 不要使用DROP/ALTER/DELETE/INSERT/UPDATE
3. 如果问题涉及数据安全, 返回拒绝原因
4. 使用适当的聚合函数和JOIN
5. 输出JSON: {{"sql": "...", "explanation": "..."}}"""

        resp = self.llm.invoke(prompt)
        import json
        result = json.loads(resp.content)
        sql = result["sql"]

        # 安全检查
        dangerous = ["DROP", "DELETE", "INSERT", "UPDATE", "ALTER", "CREATE"]
        if any(kw in sql.upper() for kw in dangerous):
            return {"error": "不允许执行非SELECT语句", "sql": sql}

        try:
            cursor = self.conn.cursor()
            cursor.execute(sql)
            rows = cursor.fetchall()
            columns = [d[0] for d in cursor.description]
            return {
                "sql": sql,
                "explanation": result["explanation"],
                "columns": columns,
                "data": rows,
                "row_count": len(rows)
            }
        except Exception as e:
            return {"error": str(e), "sql": sql}
